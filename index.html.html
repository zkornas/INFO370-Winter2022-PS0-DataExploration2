gdp_1960_total = gapminder_gdp_1960.GDP_PC.sum()
best_ratio_1960 = 100
best_perc_1960 = 0
best_value_1960 = 0
for i in range (1, 51):
    x = 1 - (i/100)
    gdp_1960_percent = gapminder_gdp_1960[(gapminder_gdp_1960.GDP_PC > gapminder_gdp_1960.GDP_PC.quantile(x))] #Get the top x% of GDP
    gdp_1960_percent_total = (gdp_1960_percent.GDP_PC).sum()                                                   #Get total GDP greater than x%
    gdp_1960_percent_owned = (100 * gdp_1960_percent_total)/gdp_1960_total                                     #Calculate percent of total GDP in 1960 is owned by top x%
    curr = abs((i + gdp_1960_percent_owned) -100)                                                              #See how close it is to 100
    if curr < best_ratio_1960:                                                                                 #Save it if it is good!+
        best_ratio_1960 = curr
        best_perc_1960 = i
        best_value_1960 = gdp_1960_percent_owned

print("top " + str(best_perc_1960) + "% owns " + str(best_value_1960) + "% of total GDP in 1960")
#top 25% owns 75.39576669722626% of total GDP in 1960
